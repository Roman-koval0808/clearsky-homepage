import { browser } from '$app/environment';

/**
 * STRICT ACTION LIBRARY (Technical Spec Doc Adherence)
 * Each Action ID represents a reusable capability, not a one-off instruction.
 */
export const ACTION_LIBRARY = {
    'ACT-REV-001': {
        title: 'Draft Review Response',
        description: 'Create a context-aware response draft for customer reviews.',
        base_priority: 2,
        default_mode: 'approval_required'
    },
    'ACT-COM-002': {
        title: 'Urgent Owner Notification',
        description: 'Send immediate SMS/Email alert to business owner.',
        base_priority: 1,
        default_mode: 'automatic'
    },
    'ACT-CRM-003': {
        title: 'Update CRM Record',
        description: 'Sync extracted customer intent or sentiment to CRM.',
        base_priority: 3,
        default_mode: 'automatic'
    },
    'ACT-SYS-004': {
        title: 'Trigger System Health Check',
        description: 'Initiate automated diagnostics based on reported anomalies.',
        base_priority: 2,
        default_mode: 'automatic'
    }
} as const;

export type ActionId = keyof typeof ACTION_LIBRARY;

export type DashboardData = {
    providers: any[];
    events: any[];
    signals: any[];
    enrichments: Record<string, any>;
    actionQueue: any[];
    approvals: any[]; 
    metrics: Record<string, number>;
};

class DashboardState {
    #data = $state<DashboardData>(DEFAULT_DATA);
    #selectedEventId = $state<string | null>(null);
    #activeBucket = $state<string | null>(null);

    constructor() {
        if (browser) {
            this.#load();
            window.addEventListener('storage', (e) => {
                if (e.key === 'clearsky_dashboard_data') this.#load();
            });
        }
    }

    #load() {
        if (browser) {
            const saved = localStorage.getItem('clearsky_dashboard_data');
            if (saved) {
                try {
                    const parsed = JSON.parse(saved);
                    if (!parsed.enrichments) parsed.enrichments = {};
                    if (!Array.isArray(parsed.approvals)) parsed.approvals = [];
                    if (parsed.providers && parsed.providers.length < 16) parsed.providers = DEFAULT_DATA.providers;

                    if (parsed.events) {
                        const seen = new Set();
                        parsed.events = parsed.events.filter((e: any) => {
                            if (seen.has(e.id)) return false;
                            seen.add(e.id);
                            return true;
                        });
                    }
                    this.#data = parsed;
                    if (!this.#selectedEventId && this.#data.events.length > 0) {
                        this.#selectedEventId = this.#data.events[0].id;
                    }
                } catch (e) { console.error(e); }
            }
        }
    }

    get data() { return this.#data; }
    get selectedEventId() { return this.#selectedEventId; }
    get activeBucket() { return this.#activeBucket; }
    set activeBucket(bucket: string | null) { this.#activeBucket = bucket; }

    get filteredEvents() {
        if (!this.#activeBucket) return this.#data.events;
        const eventIdsInBucket = new Set(this.#data.signals.filter(s => s.bucket === this.#activeBucket).map(s => s.event_id));
        return this.#data.events.filter(e => eventIdsInBucket.has(e.id));
    }

    get filteredActions() {
        if (!this.#activeBucket) return this.#data.actionQueue;
        const eventIdsInBucket = new Set(this.#data.signals.filter(s => s.bucket === this.#activeBucket).map(s => s.event_id));
        return this.#data.actionQueue.filter(a => eventIdsInBucket.has(a.event_id));
    }

    get currentEnrichment() {
        if (!this.#selectedEventId) return null;
        return this.#data.enrichments[this.#selectedEventId] || null;
    }

    selectEvent(id: string) { this.#selectedEventId = id; }

    toggleBucket(bucket: string) {
        this.#activeBucket = this.#activeBucket === bucket ? null : bucket;
    }

    #save() {
        if (browser) localStorage.setItem('clearsky_dashboard_data', JSON.stringify(this.#data));
    }

    #generateUid(prefix: string) {
        return `${prefix}_${Math.random().toString(36).substring(2, 11)}_${Date.now()}`;
    }

    /**
     * CORE REFACTOR: Strictly following the "Deterministic Control + AI Enrichment" pipeline.
     */
    addSimulationResult(rawEvent: any, extraction: any) {
        const runId = this.#generateUid('run');

        // --- STEP 1: EVENT INTAKE & AI ENRICHMENT ---
        // AI does not make decisions; it only extracts data context.
        const event = {
            id: runId,
            original_event_id: rawEvent.event_id,
            provider_id: rawEvent.provider,
            event_type: rawEvent.event_type,
            unstructured_text: rawEvent.unstructured_text,
            occurred_at: new Date().toISOString(),
            handoff_eligible: true // Determined by intake rules, not AI.
        };

        this.#data.events = [event, ...this.#data.events].slice(0, 20);
        this.#data.enrichments[runId] = {
            ...extraction,
            id: this.#generateUid('enr'),
            event_id: runId,
            raw_text: rawEvent.unstructured_text
        };

        // --- STEP 2: SIGNAL EVALUATION (DETERMINISTIC RULES ENGINE) ---
        // Rules look at the enriched event and decide if a Signal candidate fires.
        const signals = this.#evaluateSignals(event, extraction);
        this.#data.signals = [...signals, ...this.#data.signals];

        // --- STEP 3: ORCHESTRATOR DECISION (DETERMINISTIC LOGIC) ---
        // Orchestrator takes signals and maps them to strict Action Library IDs.
        this.#runOrchestrator(signals, extraction);

        this.#selectedEventId = runId;
        this.#save();
    }

    #evaluateSignals(event: any, extraction: any) {
        const signals: any[] = [];
        const baseSignal = {
            id: this.#generateUid('sig'),
            event_id: event.id,
            confidence: extraction.confidence_score,
            created_at: new Date().toISOString()
        };

        // DETERMINISTIC RULE: Complaint presence always creates a Bottleneck signal.
        if (extraction.complaint_topics?.length > 0) {
            signals.push({ ...baseSignal, name: 'Communication Bottleneck', bucket: 'Bottleneck' });
        }

        // DETERMINISTIC RULE: High urgency sentiment creates a Risk signal.
        if (extraction.urgency_level === 'high') {
            signals.push({ ...baseSignal, name: 'Urgent Intervention Required', bucket: 'Risk' });
        }

        // DETERMINISTIC RULE: Intent to purchase creates an Opportunity signal.
        if (extraction.intent?.toLowerCase().includes('quote') || extraction.intent?.toLowerCase().includes('price')) {
            signals.push({ ...baseSignal, name: 'Conversion Opportunity', bucket: 'Opportunity' });
        }

        // Fallback for Momentum
        if (signals.length === 0) {
            signals.push({ ...baseSignal, name: 'Engagement Momentum', bucket: 'Momentum' });
        }

        return signals;
    }

    #runOrchestrator(signals: any[], extraction: any) {
        for (const signal of signals) {
            let actionId: ActionId | null = null;
            let overrideMode: 'automatic' | 'approval_required' | null = null;

            // ORCHESTRATOR LOGIC: Map Signals to strict Action Library
            if (signal.bucket === 'Bottleneck') {
                actionId = 'ACT-REV-001'; // Review Response Draft
                overrideMode = 'approval_required'; // Bottlenecks always require human review
            } else if (signal.bucket === 'Risk') {
                actionId = 'ACT-COM-002'; // Urgent Notification
                overrideMode = 'automatic'; // Risks trigger alerts immediately
            } else if (signal.bucket === 'Opportunity') {
                actionId = 'ACT-CRM-003'; // Update CRM
            } else {
                actionId = 'ACT-SYS-004'; // System Health / Momentum
            }

            if (actionId) {
                const actionMeta = ACTION_LIBRARY[actionId];
                const action = {
                    id: this.#generateUid('act'),
                    action_id: actionId,
                    title: actionMeta.title,
                    owner: 'Orchestrator', // AI is not an owner.
                    confidence: signal.confidence,
                    execution_mode: overrideMode || actionMeta.default_mode,
                    priority: actionMeta.base_priority,
                    event_id: signal.event_id,
                    status: 'queued'
                };

                this.#data.actionQueue = [action, ...this.#data.actionQueue].slice(0, 10);

                // If mode is approval_required, also push to approvals list
                if (action.execution_mode === 'approval_required') {
                    this.#data.approvals = [{
                        id: this.#generateUid('app'),
                        type: action.title,
                        content: extraction.summary,
                        event_id: signal.event_id,
                        status: 'pending',
                        created_at: new Date().toISOString()
                    }, ...this.#data.approvals].slice(0, 50);
                }
            }
        }
    }

    clear() {
        this.#data = { ...DEFAULT_DATA, events: [], actionQueue: [], enrichments: {}, approvals: [] };
        this.#selectedEventId = null;
        this.#activeBucket = null;
        this.#save();
    }
}

export const dashboardState = new DashboardState();

		'system_health'
	] as const;

	type Provider = (typeof PROVIDERS)[number];

	type MockSignal = {

		name: string;
		provider: Provider;
		event_type: string;
		unstructured_text: string;

	const MOCK_SIGNALS: MockSignal[] = [

			name: 'Telnyx SMS: quote request + timeline',

			event_type: 'sms_received',

	type UiState = {

		loading: boolean;
		http_status: number | null;
		response_json: unknown | null;
		error: string | null;
	};

	type ExtractionResult = {
		intent: string;
		service_requested: string;
		problem_type: string;
		urgency_level: 'low' | 'medium' | 'high';
		timeline: string;
		sentiment: 'positive' | 'neutral' | 'negative';
		praise_topics: string[];
		complaint_topics: string[];
		summary: string;
		confidence_score: number;
	};

	let form = $state({
		event_id: '',
		provider: MOCK_SIGNALS[0].provider as Provider,
		event_type: MOCK_SIGNALS[0].event_type,
		unstructured_text: MOCK_SIGNALS[0].unstructured_text
	});

	let ui: UiState = $state({
		selected_mock_id: MOCK_SIGNALS[0].id,
		loading: false,
		http_status: null,
		response_json: null,
		error: null
	});

	const selectedMock = $derived(MOCK_SIGNALS.find((m) => m.id === ui.selected_mock_id) ?? MOCK_SIGNALS[0]);

	const extraction = $derived.by(() => {
		const value = ui.response_json as any;
		if (value?.success && value?.extraction) return value.extraction as ExtractionResult;
		return null;
	});

	const lastRunLabel = $derived.by(() => {
		if (ui.loading) return 'Running';
		if (ui.http_status === null) return 'Idle';
		if (ui.http_status >= 200 && ui.http_status < 300) return ui.http_status === 202 ? 'Accepted (no AI)' : 'Success';
		return 'Client error';
	});

	function regenerateEventId() {
		form.event_id = globalThis.crypto?.randomUUID?.() ?? `${Date.now()}-${Math.random()}`;
	}

	function loadMock(mockId: string) {
		const mock = MOCK_SIGNALS.find((m) => m.id === mockId);
		if (!mock) return;

		ui.selected_mock_id = mock.id;
		form.provider = mock.provider;
		form.event_type = mock.event_type;
		form.unstructured_text = mock.unstructured_text;

		if (!form.event_id) regenerateEventId();
	}

	function clearOutput() {
		ui.http_status = null;
		ui.response_json = null;
		ui.error = null;
	}

	import { dashboardState } from '$lib/clearsky/dashboard-state.svelte.ts';

	async function submit() {
		ui.loading = true;
		ui.error = null;
		ui.http_status = null;
		ui.response_json = null;

		try {
			const res = await fetch('/api/ai-signals/extract', {
				method: 'POST',
				headers: { 'Content-Type': 'application/json' },
				body: JSON.stringify({
					event_id: form.event_id,
					provider: form.provider,
					event_type: form.event_type,
					unstructured_text: form.unstructured_text
				})
			});

			ui.http_status = res.status;
			const json = await res.json().catch(() => null);
			ui.response_json = json;

			// Save to dashboard state if successful
			if (json?.success && json?.extraction) {
				dashboardState.addSimulationResult(
					{
						event_id: form.event_id,
						provider: form.provider,
						event_type: form.event_type,
						unstructured_text: form.unstructured_text
					},
					json.extraction
				);
			}
		} catch (err) {
			ui.error = err instanceof Error ? err.message : String(err);
		} finally {
			ui.loading = false;
		}
	}

	onMount(() => {
		regenerateEventId();
	});
</script>
